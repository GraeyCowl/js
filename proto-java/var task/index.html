Types of Arguments Unary

let a = 1;

a = -a;

console.log( a ); // -1, unary negation was applied

Types of Arguments Binary

let b = 1, c = 3;

console.log( c - b ); // 2, binary minus subtracts values

Remainder

console.log( 5 % 2 ); // 1, the remainder of 5 divided by 2

console.log( 8 % 3 ); // 2, the remainder of 8 divided by 3

console.log( 8 % 4 ); // 0, the remainder of 8 divided by 4

Exponentiation **

console.log( 2 ** 2 ); // 2² = 4

console.log( 2 ** 3 ); // 2³ = 8

console.log( 2 ** 4 ); // 2⁴ = 16

console.log( 4 ** (1/2) ); // 2 (power of 1/2 is the same as a square root)

console.log( 8 ** (1/3) ); // 2 (power of 1/3 is the same as a cubic root)
